## [15. 3Sum](https://leetcode.com/problems/3sum/)

#### java

```java
class Solution {
    public List<List<Integer>> threeSum(int[] nums) {
        List<List<Integer>> res = new ArrayList<>();
        int len = nums.length;
        Arrays.sort(nums);
        if (nums[0] > 0 || nums[len - 1] < 0) return res;
        for (int i = 0; i < len - 2; i ++) {
            if (nums[i] > 0) break;
            if (i > 0 && nums[i] == nums[i - 1]) continue;
            int j = i + 1, k = len - 1;  // 一前一后夹逼
            while (j < k) {
                if (nums[i] + nums[j] + nums[k] > 0) k --;
                else if (nums[i] + nums[j] + nums[k] < 0) j ++;
                else {
                    /*
                    List<Integer> tmp = new ArrayList<>();
                    Collections.addAll(tmp, nums[i], nums[j], nums[k]);
                    res.add(tmp);
                    */
                    res.add(List.of(nums[i], nums[j], nums[k])); // java 9 + 新特性
                    // 移动 j, k, 使得 nums[j], nums[k] 与当前不相等
                    while (j + 1 < len && nums[j + 1] == nums[j]) j ++;
                    j ++;
                    while (k - 1 > -1 && nums[k - 1] == nums[k]) k --;
                    k --;
                }
            }
        }
        return res;
    }
}
```

```c++
class Solution {

                }
            }

    }
```

```rust
use std::cmp::Ordering;
impl Solution {
    pub fn three_sum(mut nums: Vec<i32>) -> Vec<Vec<i32>> {
        let mut ans: Vec<Vec<i32>> = Vec::new();
        nums.sort();
        for i in 0..=nums.len() - 3 {
            if nums[i] > 0 { break; }
            if i > 0 && nums[i] == nums[i - 1] { continue; }
            let mut j = i + 1;
            let mut k = nums.len() - 1;
            while j < k {
                match (nums[i] + nums[j] + nums[k]).cmp(&0) {
                    Ordering::Greater => k -= 1,
                    Ordering::Less => j += 1,
                    Ordering::Equal => {
                        ans.push(vec![nums[i], nums[j], nums[k]]);
                        while j + 1 < nums.len() && nums[j] == nums[j + 1] { j += 1; }
                        j += 1;
                        while k - 1 > 0 && nums[k] == nums[k - 1] { k -= 1; }
                        k -= 1;
                    }
                }
            }
        }
        ans
    }
}
```

## [18. 4Sum](https://leetcode.com/problems/4sum/)

#### Java

```java
class Solution {
    public List<List<Integer>> fourSum(int[] nums, int target) {
        Arrays.sort(nums);
        int len = nums.length;
        List<List<Integer>> res = new ArrayList<>();
        int base = target % 4 == 0 ? target / 4 : target / 4 + 1; // 上取整
        for (int i = 0; i < len - 3; i ++) {
            if (nums[i] > base) break;
            if (i > 0 && nums[i] == nums[i - 1]) continue;
            for (int j = i + 1; j < len - 2; j ++) {
                if (nums[i] + nums[j] > 2 * base) break; // 这一层也可以剪枝
                if (j > i + 1 && nums[j] == nums[j - 1]) continue;
                
                int k = j + 1, l = len - 1;
                while (k < l) {
                    int sum = nums[i] + nums[j] + nums[k] + nums[l];
                    if (sum > target) l --;
                    else if (sum < target) k ++;
                    else {
                        res.add(List.of(nums[i], nums[j], nums[k], nums[l]));
                        while (k + 1 < len && nums[k] == nums[k + 1]) k ++;
                        k ++;
                        while (l - 1 > -1 && nums[l] == nums[l - 1]) l --;
                        l --;
                    }
                }
            }
        }
        return res;
    }
}
```

```kotlin
class Solution {
    fun fourSum(nums: IntArray, target: Int): List<List<Int>> {
        val ans = mutableListOf<List<Int>>()
        if (nums.size < 4) return ans
        nums.sort()
        val base = if(target % 4 == 0) target / 4 else target / 4 + 1
        for (i in 0 .. nums.lastIndex - 3) {
            if (nums[i] > base) break
            if (i > 0 && nums[i] == nums[i - 1]) continue
            for (j in i + 1 .. nums.lastIndex - 2) {
                if (nums[i] + nums[j] > base * 2) break
                if (j > i + 1 && nums[j - 1] == nums[j]) continue
                
                var k = j + 1; var l = nums.lastIndex
                while (k < l) {
                    val sum = nums[i] + nums[j] + nums[k] + nums[l]
                    when {
                        sum > target -> l--
                        sum < target -> k++
                        else -> {
                            ans.add(listOf(nums[i], nums[j], nums[k], nums[l]))
                            k++
                            l--
                            while (k < l && nums[k] == nums[k - 1]) k++
                            while (k < l && nums[l] == nums[l + 1]) l--
                        }
                    }
                }
            }
        }
        return ans
    }
}
```
