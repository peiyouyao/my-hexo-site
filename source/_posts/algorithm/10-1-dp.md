# Mark of Range

```
我们借用 Rust Lang 中切片的写法, 约定:
Iterator it[i..j]  -> [it[i], ... , it[j-1]]
Iterator it[i..=j] -> [it[i], ... , it[j]]
特别地, 截取 it 的前 len 长度, 即索引为 [0, 1, ..., len-1] 的部分为 it[..l]

e.g.
[1,2,3][0..=1] = [1,2]
"abc"[0..2] = "ab"
"abc"[..0] = "" 空串
```

```c++
class Solution {
public:
    int uniquePathsWithObstacles(vector<vector<int>>& obstacleGrid) {
        int val = -2;
        for (int i = 0; i < obstacleGrid.size(); i ++) {
            if (obstacleGrid[i][0] == 1) val = -1;
            obstacleGrid[i][0] = val;
        }
        val = -2;
        for (int j = 0; j < obstacleGrid[0].size(); j ++) {
            if (obstacleGrid[0][j] == 1 || obstacleGrid[0][j] == -1) val = -1;
            obstacleGrid[0][j] = val;
        }
        for (int i = 1; i < obstacleGrid.size(); i ++) {
            for (int j = 1; j < obstacleGrid[0].size(); j ++) {
                if (obstacleGrid[i][j] == 1) obstacleGrid[i][j] = -1;
                else obstacleGrid[i][j] = -1 - (- 1 - obstacleGrid[i - 1][j] - 1 - obstacleGrid[i][j - 1]);
            }
        }
        return - 1 - obstacleGrid[obstacleGrid.size() - 1][obstacleGrid[0].size() - 1];
    }
```

## [343. Integer Break](https://leetcode.com/problems/integer-break/)

#### Java

*Math*

> If an optimal product contains a factor f >= 4, then you can replace it with factors 2 and f-2 without losing optimality, as 2*(f-2) = 2f-4 >= f. So you never need a factor greater than or equal to 4, meaning you only need factors 1, 2 and 3 (and 1 is of course wasteful and you'd only use it for n=2 and n=3, where it's needed).
>
> 翻译一下: 对 $\forall n > 0$, 若 $\exist$ 划分 $\Delta$ s.t. $n = n_1 + ... + n_k$, and $n_1 \times ... \times n_k$ 最大, 若某个 $n_r >= 4$ 则它可以被拆分成 2 和 $n_r - 2$并插入原来划分, 新的划分对应的乘积也是最大值.
>
> 因为 $(n_r-2) * 2 = 2 n_r - 4 > n_r$

```java
class Solution {
    public int integerBreak(int n) {
        if (n <= 3) return n - 1; // n = 2, 3
        int q = n / 3, r = n % 3; // 每一份最多是3 故 % 3
        switch (r) {
            case 0: { return (int) Math.pow(3, q); } // 9 = 3 + 3 + 3
            case 1: { return (int) Math.pow(3, q - 1) * 4; } // 10 = 3 + 3 + 3 + 1 = 3 + 3 + 4
            default: { return (int) Math.pow(3, q) * 2; } // 11 = 3 + 3 + 3 + 2
        }
    }
}
```

*DP*

```c++
class Solution {
public:
    int integerBreak(int n) {
        vector<int> dp(n + 1, 0);
        dp[0] = 0;
        dp[1] = 1;
        int ans = 1;
        for (int i = 2; i <= n; i ++) {
            for (int j = 2; j <= i; j ++) {
                dp[i] = max(dp[i], max(j * (i - j), j * dp[i - j])); // 把 i 拆成若干份, 其中一份是 j
            }
            ans = max(ans, dp[i]);
        }
        return ans;
    }
};
```



## [96. Unique Binary Search Trees](https://leetcode.com/problems/unique-binary-search-trees/)

#### Java

```java
class Solution {
    public int numTrees(int n) {
        int[] dp = new int[n+1];
        dp[0] = 1; // 重要, 空树也要对应一种构造情况
        dp[1] = 1;
        for (int i = 2; i <= n; i ++) {
            for (int root = 1; root <= i; root ++) { // 对 root 的值进行枚举
                dp[i] += dp[root - 1] * dp[i - root];
                // 左子树的结点个数: root - 1; 右子树的结点个数: i - root
            }
        }
        return dp[n];
    }
}
```

```java
class Solution {
    private int countBST(int start, int end) {  // 计算 int:[start,end] 中BST的数量 (种类)
        int cnt = 0;
        if (start > end) {
            cnt ++;  // 空树也要算作一种情况
            return cnt;
        }
        for (int rootVal = start; rootVal <= end; rootVal ++) {
            int cntLeft = countBST(start, rootVal - 1);
            int cntRight = countBST(rootVal + 1, end);
            cnt += cntLeft * cntRight;
        }
        return cnt;
    }

    public int numTrees(int n) {
        return countBST(1, n);
    }
}
```

## [198. House Robber](https://leetcode.com/problems/house-robber/)

#### Java

完整写法1

```java
class Solution {
    public int rob(int[] nums) {
        int[][] dp = new int[nums.length][2]; // 在 nums[0..=i] 中, 拿 (dp[i][1]) 不拿 (dp[i][0]) i 的最佳状态
        dp[0][0] = 0; dp[0][1] = nums[0];
        for (int i = 1; i < nums.length; i++) {
            dp[i][0] = Math.max(dp[i-1][0], dp[i-1][1]); // 不选 i
            dp[i][1] = dp[i-1][0] + nums[i]; // 选i
        }
        return Math.max(dp[dp.length - 1][0], dp[dp.length - 1][1]);
    }
}
```

## [337. House Robber III](https://leetcode.com/problems/house-robber-iii/)

#### Java

*树形 DP*

该方法与198中的二维 DP 类似

```java
class Solution {
    public int rob(TreeNode root) {
        int[] dp = traversal(root);
        return Math.max(dp[0], dp[1]);
    }

    private int[] traversal(TreeNode root) {
        int[] dp = new int[2]; // dp[0] 不偷当前节点, 所拿到的最大金额; dp[1] 偷当前节点, 所拿到的最大金额
        if (root == null) return dp;
        int[] dpL = traversal(root.left);
        int[] dpR = traversal(root.right);
        dp[0] = Math.max(dpL[0], dpL[1]) + Math.max(dpR[0], dpR[1]); // 不偷当前节点, 则左右节点偷不偷无所谓, 因此在左右节点中选取最大的状态并加起来
        dp[1] = dpL[0] + dpR[0] + root.val; // 偷当前节点, 则左右节点只能选取不偷的状态, 加起来, 并加上当前的 val
        return dp;
    }
}
```

```c++
        if (!root) return {0, 0};
        vector<int> dpL = dfs(root->left);
        vector<int> dpR = dfs(root->right);
        vector<int> dp(2, 0);
        dp[0] = max(dpL[0], dpL[1]) + max(dpR[0], dpR[1]);
        dp[1] = dpL[0] + dpR[0] + root->val;
        return dp;
```

```c++
class Solution {
    unordered_map<TreeNode*, int> dp0, dp1;
    void dfs(TreeNode* root) {
        if (!root) return;
        dfs(root->left);
        dfs(root->right);
        dp0[root] = max(dp0[root->left], dp1[root->left]) + max(dp0[root->right], dp1[root->right]);
        dp1[root] = dp0[root->left] + dp0[root->right] + root->val;
    }

        dfs(root);
        return max(dp0[root], dp1[root]);
    }
```

*DP*

原始写法

```java
class Solution {
    private int max(int... nums) {
        int max = Integer.MIN_VALUE;
        for (int num: nums) max = Math.max(max, num);
        return max;
    }

    public int maxProfit(int[] prices) {
        int[][] dp = new int[prices.length][3];  // i 时刻的三种状态
        dp[0][0] = 0;  // 不操作
        dp[0][1] = -prices[0];  // 第一次持有
        dp[0][2] = 0;  // 第一次不持有
        for (int i = 1; i < prices.length; i++) {
            dp[i][0] = dp[i-1][0];
            dp[i][1] = max(dp[i-1][1], dp[i-1][0] - prices[i]);
            dp[i][2] = max(dp[i-1][2], dp[i-1][1] + prices[i]);
        }
        return max(dp[dp.length - 1][0], dp[dp.length - 1][1], dp[dp.length - 1][2]);
    }
}
```

## [188. Best Time to Buy and Sell Stock IV](https://leetcode.com/problems/best-time-to-buy-and-sell-stock-iv/)

#### Java

模仿上题, 直接滚动数组

```java
class Solution {
    private int max(int[] nums) {
        int max = Integer.MIN_VALUE;
        for (int num: nums) max = Math.max(max, num);
        return max;
    }

    public int maxProfit(int k, int[] prices) {
        int[] dp = new int[2*k + 1];
        // init
        for (int d = 1; d < dp.length; d ++) {  // d = 1, ..., 2k
            if (d % 2 == 1) dp[d] = -prices[0];
        }
		// calcu
        for (int i = 1; i < prices.length; i ++) {
            int[] newDp = new int[2*k + 1];
            for (int d = 1; d < dp.length; d ++) {  // 可优化
                newDp[d] = Math.max(dp[d], dp[d-1] + (d % 2 == 1 ? -1 : 1) * prices[i]);
            }
            for (int d = 0; d < dp.length; d ++) dp[d] = newDp[d];
        }
        return max(dp);
    }
}
```

```c++
        vector<int> dp(2 * k, 0);
        for (int j = 0; j < 2 * k; j ++) if ((j & 1) == 0) dp[j] = - prices[0]; // 位运算优先级极低, 记得打括号
        for (int i = 1; i < prices.size(); i ++) {

            dp[0] = max(dp[0], - prices[i]);
        }
        int ans = 0;
        for (int d : dp) ans = max(ans, d);
        return ans;
    }
};
```

## [309. Best Time to Buy and Sell Stock with Cooldown](https://leetcode.com/problems/best-time-to-buy-and-sell-stock-with-cooldown/)

Buy and Sell Stock II 变形, 需要维护至少两天的状态

#### Java

```java
class Solution {
    public int maxProfit(int[] prices) {
        if (prices.length < 2) return 0;
        if (prices.length == 2) return Math.max(0, prices[1] - prices[0]);
        int[][] dp = new int[prices.length][2];
        dp[0][0] = 0;  // 未持有
        dp[0][1] = -prices[0];  // 持有
        dp[1][0] = Math.max(dp[0][0], dp[0][1] + prices[1]);
        dp[1][1] = Math.max(dp[0][1], -prices[1]);  // 如果第1天持有股票, 他不可能是第0天卖出后买入, 只可能是第1天买入, 或第1天不操作, 延续第0天
        for (int i = 2; i < prices.length; i ++) {
            dp[i][0] = Math.max(dp[i-1][0], dp[i-1][1] + prices[i]);  // 卖
            dp[i][1] = Math.max(dp[i-1][1], dp[i-2][0] - prices[i]);  // 买
        }
        return Math.max(dp[dp.length - 1][0], dp[dp.length - 1][1]);
    }
}
```

## [714. Best Time to Buy and Sell Stock with Transaction Fee](https://leetcode.com/problems/best-time-to-buy-and-sell-stock-with-transaction-fee/)

Buy and Sell Stock II 变形, 需要在买入时 (持有时) 加入费用. 

为什么不能在更新未持有状态时刻缴费? 因为这里的未持有状态其实是压缩了的: 它可以代表一开始未持有, 也可以代表卖出股票后未持有

#### Java

```java
class Solution {
    public int maxProfit(int[] prices, int fee) {
        int[] dp = new int[2];
        dp[0] = 0;  // 未持有
        dp[1] = -prices[0] - fee;  // 持有; 统一在买入时缴费
        for (int i = 1; i < prices.length; i ++) {
            int t0 = Math.max(dp[0], dp[1] + prices[i]);
            int t1 = Math.max(dp[1], dp[0] - prices[i] - fee);
            dp[0] = t0; dp[1] = t1;
        }
        return Math.max(dp[0], dp[1]);
    }
}
```

```c++
    int maxProfit(vector<int>& prices, int fee) {
        int dp[3] = {0, - prices[0], - fee};
        // dp[0] - 未操作, dp[1] - 持有状态, dp[2] - (操作以后) 未持有状态

            int t0 = dp[0];
            int t1 = max(dp[1], max(dp[0] - prices[i], dp[2] - prices[i]));
            int t2 = max(dp[2], dp[1] + prices[i] - fee);
            dp[0] = t0; dp[1] = t1; dp[2] = t2;
        }
        return max(dp[0], max(dp[1], dp[2]));
    }
};
```
