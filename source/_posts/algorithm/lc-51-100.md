## [57. Insert Interval](https://leetcode.com/problems/insert-interval/)

```java
class Solution {
    public int[][] insert(int[][] intervals, int[] newInterval) {
        int pos = Arrays.binarySearch(intervals, newInterval, (itv1, itv2) -> itv1[0] - itv2[0]);
        int insertPos = (pos < 0) ? -pos-1 : pos;
        List<int[]> ans = new ArrayList<>();
        for (int i = 0; i < insertPos; i ++) ans.add(intervals[i]); // pos 之前的原封不动
        for (int i = insertPos - 1; i < intervals.length; i ++) {
            int[] insertItv = (i == insertPos - 1) ? newInterval : intervals[i];
            if (ans.isEmpty()) ans.add(insertItv);
            else {
                int[] last = ans.getLast();
                if (last[1] < insertItv[0]) ans.add(insertItv);
                else last[1] = Math.max(last[1], insertItv[1]);
            }
        }
        return ans.toArray(new int[0][]);
    }
}
```

## [58. Length of Last Word](https://leetcode.com/problems/length-of-last-word/) [Array]

```java
class Solution {
    public int lengthOfLastWord(String s) {
        int tail = s.length() - 1;
        while ( tail > -1 && s.charAt(tail) == ' ' ) tail --;
        if ( tail == -1 ) return 0;
        int head = tail;
        while ( head > -1 && s.charAt(head) != ' ' ) head --;
        return tail - head;
    }
}
```

## [60. Permutation Sequence](https://leetcode.com/problems/permutation-sequence/)

### 法一

联系 [31. Next Permutation](https://leetcode.com/problems/next-permutation/)

```java
class Solution {
    private void swap(int[] nums, int i, int j) {
        int t = nums[i]; nums[i] = nums[j]; nums[j] = t;
    }

    private void reverse(int[] nums, int i, int j) {
        while (i < j) {
            swap(nums, i, j); i ++; j --;
        }
    }

    private void nextPermutation(int[] nums) {
        int len = nums.length;
        int breakIdx = len - 2;
        while (breakIdx > -1) {
            if (nums[breakIdx] < nums[breakIdx + 1]) break;
            breakIdx --;
        }

        if (breakIdx == -1) {
            reverse(nums, 0, len - 1);
            return;
        }

        int breakVal = nums[breakIdx];
        int swapIdx = breakIdx + 1;
        int swapVal = nums[swapIdx];
        for (int i = breakIdx + 1; i < len; i ++) {
            if (nums[i] <= breakVal) continue;
            if (nums[i] <= swapVal) {
                swapIdx = i;
                swapVal = nums[i];
            }
        }
        swap(nums, breakIdx, swapIdx);
        reverse(nums, breakIdx + 1, len - 1);
    }

    public String getPermutation(int n, int k) {
        int[] nums = new int[n];
        for (int i = 0; i < n; i ++) nums[i] = i + 1;
        while (k - 1 > 0) {
            nextPermutation(nums);
            k --;
        }
        char[] chs = new char[n];
        for (int i = 0; i < n; i ++) chs[i] = (char)(nums[i] + '0');
        return new String(chs);
    }
}
```

```kotlin
class Solution {
    private fun genPermutation(nums: IntArray): Unit {
        fun swap(i: Int, j: Int) { val t = nums[i]; nums[i] = nums[j]; nums[j] = t }
        fun reverseRange(i: Int, j: Int): Unit { // kotlin 中, 函数参数都是 val 的
            var ii = i
            var jj = j
            while (ii < jj) {
                swap(ii, jj)
                ii ++
                jj --
            }
        }

        var breakIdx = nums.lastIndex - 1
        while (breakIdx > -1 && nums[breakIdx] > nums[breakIdx + 1]) breakIdx --
        if (breakIdx == -1) {
            reverseRange(breakIdx+1, nums.lastIndex)
            return
        }

        val breakVal = nums[breakIdx]
        var swapIdx = breakIdx + 1
        var swapVal = nums[swapIdx]
        for (i in breakIdx + 1 .. nums.lastIndex) {
            if (nums[i] < breakVal) continue
            if (nums[i] < swapVal) {
                swapVal = nums[i]
                swapIdx = i
            }
        }
        swap(breakIdx, swapIdx)
        reverseRange(breakIdx+1, nums.lastIndex)
    }

    fun getPermutation(n: Int, k: Int): String {
        val nums = IntArray(n) { it -> it + 1 }
        repeat(k-1) { genPermutation(nums) }
        return nums.joinToString("")
    }
}
```

### 法二

回溯, 使用 boolean 返回值的回溯函数来终止递归 (当找到了第 k 个结果)

```java
class Solution {
    private String ans;
    private int cnt = 0;
    private boolean backtrack(char[] path, boolean[] used, int th, int k) {
        if (th == path.length) {
            cnt ++;
            if (k == cnt) {
                ans = new String(path);
                return true;
            } else return false;
        }
        for (int i = 0; i < used.length; i++) {
            if (used[i]) continue;
            used[i] = true;
            path[th] = (char)(i + 1 + '0');
            if (backtrack(path, used, th + 1, k)) return true;
            path[th] = '\0';
            used[i] = false;
        }
        return false;
    }

    public String getPermutation(int n, int k) {
        char[] path = new char[n];
        boolean[] used = new boolean[n];
        backtrack(path, used, 0, k);
        return ans;
    }
}
```

```kotlin
class Solution {
    fun getPermutation(n: Int, k: Int): String {
        var ans: String = ""
        var cnt = 0
        val path = mutableListOf<Int>()
        val used = BooleanArray(n) {false}
        fun backtrack(path: MutableList<Int>, k: Int, used: BooleanArray, n: Int): Boolean {
            if (path.size == n) {
                cnt ++
                if (cnt == k) {
                    ans = path.joinToString("")
                    return true
                } else return false
            }

            for (i in 1 .. n) {
                if (used[i-1]) continue
                used[i-1] = true
                path.add(i)
                if (backtrack(path, k, used, n)) return true
                path.removeLast()
                used[i-1] = false
            }
            return false
        }
        backtrack(path, k, used, n)
        return ans
    }
}
```

## [61. Rotate List](https://leetcode.com/problems/rotate-list/)

```java
class Solution {
    private ListNode rotateOnce(ListNode head) {
        ListNode p1 = head, p2 = head.next;
        while (p2.next != null) {
            p1 = p1.next; p2 = p2.next;
        }
        p2.next = head;
        p1.next = null;
        return p2;
    }

    public ListNode rotateRight(ListNode head, int k) {
        if (head == null || head.next == null) return head;
        int size = 0;
        ListNode p = head;
        while (p != null) {
            size ++;
            p = p.next;
        }
        k %= size;
        while (k > 0) {
            head = rotateOnce(head);
            k --;
        }
        return head;
    }
}
```

```kotlin
            p = p.next

        p = head

            p = p?.next
            q = q?.next
        }
        var ans: ListNode? = p?.next
        p?.next = null
        q?.next = head
        return ans
    }
}
```

## [64. Minimum Path Sum](https://leetcode.com/problems/minimum-path-sum/)

DP

```java
class Solution {
    public int minPathSum(int[][] grid) {
        int m = grid.length, n = grid[0].length;
        int[][] dp = new int[m][n];
        int sum = 0;
        for (int i = 0; i < m; i ++) {
            sum += grid[i][0];
            dp[i][0] = sum;
        }
        sum = 0;
        for (int j = 0; j < n; j ++) {
            sum += grid[0][j];
            dp[0][j] = sum;
        }

        for (int i = 1; i < m; i ++) {
            for (int j = 1; j < n; j ++) {
                dp[i][j] = grid[i][j] + Math.min(dp[i - 1][j], dp[i][j - 1]);
            }
        }
        return dp[m - 1][n - 1];
    }
}
```

```kotlin
class Solution {
    fun minPathSum(grid: Array<IntArray>): Int {
        val m = grid.size
        val n = grid[0].size
        val dp = Array<IntArray>(m) { IntArray(n) }
        for (i in 0 ..< m) {
            for (j in 0 ..< n) {
                dp[i][j] = when {
                    i == 0 && j == 0 -> grid[i][j]
                    i == 0 && j > 0 -> dp[i][j-1] + grid[i][j]
                    i > 0 && j == 0 -> dp[i-1][j] + grid[i][j]
                    else -> minOf(dp[i-1][j], dp[i][j-1]) + grid[i][j]
                }
            }
        }
        return dp[m-1][n-1]
    }
}
```

## [66. Plus One](https://leetcode.com/problems/plus-one/)

这种题目的要点就是: 计算和进位分离

```java
class Solution {
    public int[] plusOne(int[] digits) {
        digits[digits.length - 1] += 1;
        LinkedList<Integer> res = new LinkedList<>();
        for (int i = digits.length - 1; i > -1; i --) {
            int curVal = digits[i];
            if (curVal < 10) res.addFirst(curVal);
            else {
                if (i != 0) {
                    res.addFirst(curVal % 10);
                    digits[i - 1] += curVal / 10;
                } else {
                    res.addFirst(curVal % 10);
                    res.addFirst(curVal / 10);
                }
            }
        }
        return res.stream().mapToInt(Integer::intValue).toArray();
    }
}
```

```java
        digits[digits.length - 1] += 1;  // 计算
```

```java
class Solution {
    public String addBinary(String a, String b) {
        int[] tmp = new int[Math.max(a.length(), b.length())];
        for (int i = 0; i < tmp.length; i ++) {  // 先计算
            int idxA = a.length() - 1 - i, idxB = b.length() - 1 - i, idxT = tmp.length - 1 - i;
            int numA = (idxA < 0) ? 0 : (int)(a.charAt(idxA) - '0');
            int numB = (idxB < 0) ? 0 : (int)(b.charAt(idxB) - '0');
            tmp[idxT] = numA + numB;
        }
        

        for (int i = tmp.length - 1; i > 0; i --) {  // 再进位
            tmp[i - 1] += tmp[i] / 2;
            tmp[i] %= 2;
        }

        if (tmp[0] == 2) tmp[0] = 10;
        else if (tmp[0] == 3) tmp[0] = 11;

        return Arrays.stream(tmp).mapToObj(String::valueOf).collect(Collectors.joining(""));
    }
}
```

```java
class Solution {
    public String simplifyPath(String path) {
        String[] paths = path.split("/");
        LinkedList<String> stk = new LinkedList<>();
        stk.addLast("");
        for (String s: paths) {
            if (".".equals(s)) continue;
            if ("".equals(s)) continue;
            if ("..".equals(s)) {
                if (stk.size() == 0) {
                    stk.addLast("");
                    continue;
                }
                stk.removeLast();
                continue;
            }
            stk.addLast(s);
        }
        String res = stk.stream().collect(Collectors.joining("/"));
        if (res.length() == 0 || res.charAt(0) != '/') return "/" + res;
        return res;
    }
}
```

## [73. Set Matrix Zeroes](https://leetcode.com/problems/set-matrix-zeroes/)

```java
class Solution {
    public void setZeroes(int[][] matrix) {
        int rows = matrix.length, cols = matrix[0].length;
        List<int[]> record = new ArrayList<>();
        for (int i = 0; i < rows; i ++) {
            for (int j = 0; j < cols; j ++) {
                if (matrix[i][j] == 0) record.add(new int[] {i, j});
            }
        }

        for (int[] index: record) {
            int row = index[0], col = index[1];
            for (int[] r: matrix) r[col] = 0;
            for (int j = 0; j < cols; j ++) matrix[row][j] = 0;
        }
    }
}
```

```java
class Solution {
    private int[][] mat;
    private int val(int th) { return mat[th / mat[0].length][th % mat[0].length]; }
    public boolean searchMatrix(int[][] matrix, int target) {
        mat = matrix;
        int l = 0, r = mat.length * mat[0].length - 1;
        while (l <= r) {
            int m = (l + r) / 2;
            if (val(m) > target) r = m - 1;
            else if (val(m) < target) l = m + 1;
            else return true;
        }
        return false;
    }
```

```java
class Solution {
    private int nr, nc;
    private char[][] board;
    private String word;
    private boolean[][] used;

    private boolean backtrack(StringBuilder sb, int i, int j) {  // 从 (i,j) 出发, 能否搜到
        used[i][j] = true;
        sb.append(board[i][j]);
        if (word.equals(sb.toString())) return true;  // 这里太划不来, 当比较到第 k 位不相等就可以返回 false
        if (i != nr-1 && !used[i+1][j]) if (backtrack(sb, i+1, j)) return true;
        if (i != 0    && !used[i-1][j]) if (backtrack(sb, i-1, j)) return true;
        if (j != nc-1 && !used[i][j+1]) if (backtrack(sb, i, j+1)) return true;
        if (j != 0    && !used[i][j-1]) if (backtrack(sb, i, j-1)) return true;
        used[i][j] = false;
        sb.deleteCharAt(sb.length()-1);
        return false;
    }

    public boolean exist(char[][] board, String word) {
        this.nr = board.length; 
        this.nc = board[0].length;
        this.board = board;
        this.word = word;
        this.used = new boolean[nr][nc];
        for (int i = 0; i < nr; i ++) {
            for (int j = 0; j < nc; j ++) {
                if (board[i][j] == word.charAt(0)) {
                    if (backtrack(new StringBuilder(), i, j)) return true;
                }
            }
        }
        return false;
    }
}
```

```java
class Solution {
    private int nr, nc;
    private char[][] board;
    private String word;
    private boolean[][] used;

    private boolean backtrack(int th, int i, int j) {  // 从 (i,j) 出发, 能否搜到
        if (word.charAt(th) != board[i][j]) return false;  // 不相等就立刻返回 false, 不磨叽
        if (th == word.length()-1) {
            return word.charAt(th) == board[i][j];
        }
        used[i][j] = true;
        if (i != nr-1 && !used[i+1][j]) if (backtrack(th+1, i+1, j)) return true;
        if (i != 0    && !used[i-1][j]) if (backtrack(th+1, i-1, j)) return true;
        if (j != nc-1 && !used[i][j+1]) if (backtrack(th+1, i, j+1)) return true;
        if (j != 0    && !used[i][j-1]) if (backtrack(th+1, i, j-1)) return true;
        used[i][j] = false;
        return false;
    }

    public boolean exist(char[][] board, String word) {
        this.nr = board.length; 
        this.nc = board[0].length;
        this.board = board;
        this.word = word;
```

```java
class Solution {
    private int[] nums;
    private int t;
    private boolean recursion(int l, int r) {
        while (l+1 <= nums.length-1 && nums[l+1] == nums[l]) l ++;  // 去重
        while (r-1 >= 0 && nums[r-1] == nums[r]) r --;  // 去重

        if (nums[l]==t) return true;
        if (nums[r]==t) return true;
        if (l >= r || l+1 == r) return false;
        int m = (l+r)/2;
        if (nums[m]==t) return true;
        if (nums[l] <= nums[m]) {  // nums:[l,m]  递增
            if (nums[l] <= t && t <= nums[m]) return recursion(l, m-1);
            else return recursion(m,r);
        } else {  // nums:[m,r] 递增
            if (nums[m] <= t && t <= nums[r]) return recursion(m+1, r);
            else return recursion(l,m);
        }
    }

    public boolean search(int[] nums, int target) {
        this.nums = nums;
        this.t = target;
        return recursion(0, nums.length-1);
    }
}
```

```java
class Solution {
    public void merge(int[] nums1, int m, int[] nums2, int n) {
        if (n==0) return;
        if (m==0) {
            for (int i = 0; i < m + n; i ++) nums1[i] = nums2[i];
            return;
        }
        int[] tmp = new int[m+n];
        int idx1 = 0, idx2 = 0;
        for (int i = 0; i < m+n; i ++) {
            if (idx1 == m && idx2 == n) continue;
            else if (idx1 == m) {
                tmp[i] = nums2[idx2];
                idx2 ++;
            } else if (idx2 == n) {
                tmp[i] = nums1[idx1];
                idx1 ++;
            } else {
                if (nums1[idx1] < nums2[idx2]) {
                    tmp[i] = nums1[idx1];
                    idx1 ++;
                } else {
                    tmp[i] = nums2[idx2];
                    idx2 ++;
                }
            }
        }
        for (int i = 0; i < m + n; i ++) nums1[i] = tmp[i];
    }
```

```java
        if (n == 1) return new String[] {"0", "1"};
        int len = (int)Math.pow(2, n - 1);
        String[] res = new String[2 * len];
        String[] former = genCode(n - 1);
        if (n % 2 == 0) {
            for (int i = 0; i < len; i ++) {
                int idx = i;
                res[i] = "0" + former[idx];
            }

            for (int i = len; i < 2 * len; i ++) {
                int idx = len - (i - len) - 1;
                res[i] = "1" + former[idx];

                String head = former[idx].substring(0, (n-1)/2);

                String tail = former[idx].substring((n-1)/2);

        }
```

**另一种思路**

> 设 n 阶格雷码集合为 G(n)，则 G(n+1) 阶格雷码可以通过以下三步得到：
>
> 给 G(n) 阶格雷码每个元素二进制形式前面添加 0，得到 G′(n)；
>
> 设 G(n) 集合倒序（镜像）为 R(n)，给 R(n) 每个元素二进制形式前面添加 1，得到 R′(n)；
>
> G(n+1)=G′(n)∪R′(n) 拼接两个集合即可得到下一阶格雷码。
>
> 由于最高位前默认为 0，因此 G′(n)=G(n)，只需在 res(即 G(n) )后添加 R′(n) 即可；
>
> 计算 R′(n)：执行 head = 1 << i 计算出对应位数，以给 R(n) 前添加 1 得到对应 R′(n)；
>
> 倒序遍历 res(即 G(n) )：依次求得 R'(n) 各元素添加至 res 尾端，遍历完成后 res(即 G(n+1))。

```cpp
class Solution {
public:
    vector<int> grayCode(int n) {
        vector<int> ans;
        ans.push_back(0);

        int head = 1;
        for (int i = 0; i < n; i ++) {
            for (int j = ans.size() - 1; j > -1; j --) {
                ans.push_back(head + ans[j]);
            }
            head <<= 1;
        }
        return ans;
    }
};
```

```go
func grayCode(n int) []int {
    var ans = make([]int, 1) 
    ans[0] = 0
    var head = 1 // head 表示当前位数的值，它会根据每一轮的格雷编码位数不断左移
    
    // 外层循环，控制格雷编码的位数，从 0 到 n-1，生成 n 位的格雷编码
    for i := 0; i < n; i++ {
        // sz 记录当前格雷编码序列的长度
        var sz = len(ans)
        
        // 内层循环，将现有的编码值和当前位数的值进行合并，生成新的编码
        for j := sz - 1; j > -1; j-- {
            // 在 ans 列表的末尾添加新编码值
            // head + ans[j] 相当于将现有的编码值与当前位进行合并
            ans = append(ans, head + ans[j])
        }
        
        // 将 head 左移一位，准备处理下一位的编码
        // head 相当于每一轮格雷编码中需要加入的 "1 << i" 部分
        head <<= 1
    }
    
    // 返回最终生成的格雷编码序列
    return ans
}
```

```java
class Solution {
    public List<Integer> grayCode(int n) {
        List<Integer> ret = new ArrayList<>(); // 创建一个ArrayList来存储最终的格雷编码序列
        ret.add(0); // 初始化格雷编码序列的第一个元素 0
        
        // 外层循环，i表示当前正在处理的是第i位的格雷编码
        // 从1到n，生成每一位的格雷编码
        for (int i = 1; i <= n; i++) {
            // sz为当前序列的大小，表示已有的格雷编码数量
            int sz = ret.size();
            
            // 内层循环，倒着遍历已有的编码
            // 通过倒着遍历，保证我们新加入的格雷编码能与原编码列表进行合并
            // 内层循环将已有的格雷编码每个数与当前的位进行合并
            for (int j = sz - 1; j > -1; j--) {
                // 将当前已有的编码与当前位的数值进行“或”运算，
                // 通过“或”操作将该位设置为1，形成新的格雷编码
                // (1 << (i - 1)) 计算出该位置的二进制数值，i从1开始，所以要用 (i - 1) 来表示第i位
                ret.add(ret.get(j) | (1 << (i - 1)));
            }
        }
        return ret;
    }
}
```

```java
class Solution {
    private ListNode reverse(ListNode head) {
        if (head == null || head.next == null) return head;
        ListNode rest = head.next;
        rest = reverse(rest);
        head.next = null;
        ListNode p = rest;
        while (p != null && p.next != null) p = p.next;
        p.next = head;
        return rest;
    }

    public ListNode reverseBetween(ListNode head, int left, int right) {
        ListNode dummy = new ListNode(0, head);
        ListNode l = dummy, r = dummy;
        for (int i = 0; i < left - 1; i ++) l = l.next;
        for (int i = 0; i < right; i ++) r = r.next;
        ListNode rest = r.next;
        r.next = null;
        ListNode reversed = reverse(l.next);
        l.next = reversed;
        r = dummy;
        for (int i = 0; i < right; i ++) r = r.next;
        r.next = rest;
        return dummy.next;
    }
}
```

```java
class Solution {
    public boolean isInterleave(String s1, String s2, String s3) {
        if (s1.length() + s2.length() != s3.length()) return false;
        boolean[][] dp = new boolean[s1.length()+1][s2.length()+1];
        for (int i = 0; i <= s1.length(); i ++) {
            for (int j = 0; j <= s2.length(); j ++) {
                if (i == 0 || j == 0) {
                    if (i == j) dp[i][j] = true;
                    else if (i == 0) dp[i][j] = s3.charAt(j - 1) == s2.charAt(j - 1) && dp[i][j - 1];
                    else dp[i][j] = s3.charAt(i - 1) == s1.charAt(i - 1) && dp[i - 1][j];
                } else {
                    char c1 = s1.charAt(i - 1), c2 = s2.charAt(j - 1), c3 = s3.charAt(i + j - 1);
                    if (c1 == c3) dp[i][j] = dp[i][j] || dp[i - 1][j];
                    if (c2 == c3) dp[i][j] = dp[i][j] || dp[i][j - 1];
                }
            }
        }
        return dp[s1.length()][s2.length()];
    }
```

```
这种定义方法的错误在于: 
0, 不便于记录 pre 节点
1, 测试样例中有 val = Integer.MIN_VALUE 的值
2, 即使解决了以上 bug, 由于是在每一个递归函数中定义的变量, 递归树的每一个"节点"独一份
   因此, 当 traversal(root.left, preVal) 执行完成后, 并不会修改当前节点的 preVal
```

```kotlin
class Solution {
    fun recoverTree(root: TreeNode?): Unit {
        val rec = mutableListOf<TreeNode?>()
        var prev: TreeNode? = null
        fun traversal(curr: TreeNode?): Unit {
            if (curr == null) return
            traversal(curr?.left)
            if (prev?.`val` ?: Int.MIN_VALUE > curr.`val`) {
                rec.add(prev)
                rec.add(curr)
            }
            prev = curr
            traversal(curr?.right)
        }
        traversal(root)
        var tmp = rec.first()?.`val`
        rec.first()?.`val` = rec.last()?.`val`
        rec.last()?.`val` = tmp
    }
}
```
