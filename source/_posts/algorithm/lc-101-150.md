## [103. Binary Tree Zigzag Level Order Traversal](https://leetcode.com/problems/binary-tree-zigzag-level-order-traversal/)

```java
class Solution {
    private void traversal(List<List<Integer>> ans, int deep, TreeNode root) {
        if (root == null) return;
        else deep ++;
        if (ans.size() < deep) ans.add(new LinkedList<Integer>());
        List<Integer> lvl = ans.get(deep - 1);
        if ((deep & 1) == 0) {
            lvl.addFirst(root.val);
        } else {
            lvl.addLast(root.val);
        }
        traversal(ans, deep, root.left);
        traversal(ans, deep, root.right);
    }
    public List<List<Integer>> zigzagLevelOrder(TreeNode root) {
        List<List<Integer>> ans = new ArrayList<>();
        traversal(ans, 0, root);
        return ans;
    }
}
```

## [109. Convert Sorted List to Binary Search Tree](https://leetcode.com/problems/convert-sorted-list-to-binary-search-tree/)

```java
class Solution {
    private TreeNode buildBST(ListNode head) {
        if (head == null) return null;
        if (head.next == null) return new TreeNode(head.val);
        ListNode slow = head, fast = head.next, pre = null;
        while (fast != null && fast.next != null) {
            pre = slow;  // pre 是跟屁虫
            slow = slow.next;
            fast = fast.next.next;
        }
        ListNode rightList = slow.next;

        ListNode leftList = null;
        if (pre != null) {
            pre.next = null;
            leftList = head;
        }
        
        TreeNode res = new TreeNode(slow.val);
        
        res.left = buildBST(leftList);
        res.right = buildBST(rightList);
        return res;
    }

    public TreeNode sortedListToBST(ListNode head) {
        return buildBST(head);
    }
}
```

## [111. Minimum Depth of Binary Tree](https://leetcode.com/problems/minimum-depth-of-binary-tree/) [Tree]

```java
class Solution {
    public int minDepth(TreeNode root) {
        int res = 0;
        if (root == null) return res;
        LinkedList<TreeNode> q = new LinkedList<>();
        q.addLast(root);
        while ( !q.isEmpty() ) {
            res ++;
            int lvlSize = q.size();
            for (int i = 0; i < lvlSize; i++) {
                TreeNode cur = q.removeFirst();
                if (cur.left == null && cur.right == null) return res;
                else {
                    if (cur.left != null) q.addLast(cur.left);
                    if (cur.right != null) q.addLast(cur.right);
                }
            }
        }
        return res;
    }
}
```

## [113. Path Sum II](https://leetcode.com/problems/path-sum-ii/) [Tree]

```java
class Solution {
    private List<List<Integer>> res = new LinkedList<>();
    private List<Integer> path = new LinkedList<>();
    private int curSum = 0;
    private void dfs(TreeNode node, int targetSum, int curSum) {
        if (node == null) return;
        curSum += node.val;
        path.addLast(node.val);
        if (node.left == null && node.right == null) {
            if (curSum == targetSum) res.addLast(new LinkedList<Integer>(path));
        } else {
            if (node.left != null) dfs(node.left, targetSum, curSum);
            if (node.right != null) dfs(node.right, targetSum, curSum);
        }
        curSum -= node.val;  // no matter whether curSum == targetSum, we need backtrack
        path.removeLast();  // no matter whether curSum == targetSum, we need backtrack
        return;
    }

    public List<List<Integer>> pathSum(TreeNode root, int targetSum) {
        dfs(root, targetSum, 0);
        return res;
    }
}
```

```cpp
class Solution {
public:
    void flatten(TreeNode* root) {
        if (root == nullptr) return;
        TreeNode* left = root->left;
        TreeNode* right = root->right;
        root->left = nullptr;
        root->right = left;
        TreeNode *p = root;
        while (p != nullptr && p->right != nullptr) p = p->right;
        p->right = right;
        p = nullptr;
        flatten(root->right);
    }
};
```

```java
class Solution {
```

```java
        List<List<Integer>> ans = new ArrayList<>(numRows);
        for (int i = 0; i < numRows; i ++) {
            ans.add(new ArrayList<>());
            List<Integer> level = ans.get(i);
            for (int j = 0; j <= i; j ++) {
                level.add(0);
                if (j == 0 || j == i) level.set(j, 1);
                else {
                    List<Integer> lastLevel = ans.get(i - 1);
                    int val = lastLevel.get(j - 1) + lastLevel.get(j);
                    level.set(j, val);
                }
            }
        }
        return ans;
    }
}
```

```java
class Solution {
    private int res = Integer.MAX_VALUE;

    private void traversal(List<List<Integer>> triangle, int lvlth, int parentidx, int curSum) {
        if (triangle.size() == lvlth) {
            res = Math.min(res, curSum);
            return;

        List<Integer> curlvl = triangle.get(lvlth);
        for (int i = parentidx; i <= Math.min(curlvl.size() - 1, parentidx + 1); i ++) {
            traversal(triangle, lvlth+1, i, curSum + curlvl.get(i));
        }
    }

        traversal(triangle, 0, 0, 0);
        return res;
    }
}
```

## [124. Binary Tree Maximum Path Sum](https://leetcode.com/problems/binary-tree-maximum-path-sum/)

可以先做 [543. Diameter of Binary Tree](https://leetcode.com/problems/diameter-of-binary-tree/) 

**错误解法**

以下代码不符合题意, 它可以算出一个二叉树中, 连通子树的最大和

```java
class Solution {
    private int max(int... nums) {
        int max = Integer.MIN_VALUE;
        for (int num: nums) max = Math.max(num, max);
        return max;
    }
    private int[] traversal(TreeNode root) {
        int[] dp = new int[2]; // dp[0] 不选择当前节点加入子树的最大和, dp[1] 选择当前节点 ...
        if (root == null) {
            dp[0] = -Integer.MIN_VALUE / 2;
            dp[1] = -Integer.MIN_VALUE / 2;
        } else {
            int[] left = traversal(root.left);
            int[] right = traversal(root.right);
            dp[0] = max(left[0], left[1], right[0], right[1]);
            dp[1] = max(root.val, root.val + max(left[1], right[1], left[1] + right[1]));
        }
        return dp;
    }
    
    public int maxPathSum(TreeNode root) {
        int[] res = traversal(root);
        return max(res[0], res[1]);
    }
}
```

**正确解法**

```
递归计算节点的最大贡献值：
对每个节点，递归计算它的左右子树的最大贡献值。如果一个子树的贡献值小于 0，就不选取它（即取 0）。
通过递归 maxGain 函数，返回节点的最大贡献值：node.val + max(leftGain, rightGain)。这个值代表了从当前节点出发，可以向下传递给父节点的最大路径和。

更新最大路径和：
在每个节点，计算通过该节点的路径和（node.val + leftGain + rightGain），并更新全局的 maxSum，确保它保持最大值。

返回最大贡献值：
每次递归返回的值是当前节点的最大贡献值，它将被传递给父节点作为右子树或左子树的最大路径和。
```



```java
class Solution {
    public int maxPathSum(TreeNode root) {
        maxGain(root);
        return maxSum;
    }
	private int maxSum = Integer.MIN_VALUE;
    private int maxGain(TreeNode node) {
        if (node == null) {
            return 0;
        }
        // 递归计算左右子节点的最大贡献值
        // 只有在最大贡献值大于 0 时，才会选取对应子节点
        int leftGain = Math.max(maxGain(node.left), 0);
        int rightGain = Math.max(maxGain(node.right), 0);
        // 节点的最大路径和取决于该节点的值与该节点的左右子节点的最大贡献值
        int priceNewpath = node.val + leftGain + rightGain;
        // 更新答案
        maxSum = Math.max(maxSum, priceNewpath);
        // 返回节点的最大贡献值
        return node.val + Math.max(leftGain, rightGain);
    }
}
```

```go
func max(nums ...int) int {
    var max = math.MinInt
    for _, v := range nums {
        if v > max {
            max = v
        }
    }
    return max
}

func maxGain(node *TreeNode, sum *int) int {
    if node == nil {
        return 0
    }

    var leftGain = max(maxGain(node.Left, sum), 0)
    var rightGain = max(maxGain(node.Right, sum), 0)

    *sum = max(*sum, leftGain + rightGain + node.Val)
    return max(leftGain, rightGain) + node.Val // 要构成路径, 左右子树只能选一边
}
```

## [125. Valid Palindrome](https://leetcode.com/problems/valid-palindrome/)

```java
class Solution {
    public boolean isPalindrome(String s) {
        int l = 0, r = s.length()-1;
        while (l < r) {
            char lch = s.charAt(l), rch = s.charAt(r);
            if ('A' <= lch && lch <= 'Z') lch = (char)(lch - 'A' + 'a');
            if ('A' <= rch && rch <= 'Z') rch = (char)(rch - 'A' + 'a');
            
            if (!('a' <= lch && lch <= 'z') && !('0' <= lch && lch <= '9')) {
                l ++;
                continue;
            }

            if (!('a' <= rch && rch <= 'z') && !('0' <= rch && rch <= '9')) {
                r --;
                continue;
            }

            if (rch != lch) return false;
            l ++; r --;
        }
        return true;
    }
}
```

```go
func isPalindrome(s string) bool {
    var valid = func(c byte) (bool, byte) {
        if '0' <= c && c <= '9' || 'a' <= c && c <= 'z' {
            return true, c
        }

        if 'A' <= c && c <= 'Z' {
            return true, c + ' '
        }

        return false, ' '
    }

    var l, r = 0, len(s) - 1
    for l < r {
        vl, cl := valid(s[l])
        vr, cr := valid(s[r])
        if !vl {
            l ++
        } else if !vr {
            r --
        } else {
            if cl == cr {
                l ++
                r --
            } else {
                return false
            }
        }
    }
    return true
}
```

## [128. Longest Consecutive Sequence](https://leetcode.com/problems/longest-consecutive-sequence/)

这道题关键难在时间复杂度要求 $O(n)$, 由于几乎所有的排序算法, 时间复杂度都大于 $O(n)$, 故该题不能简单地使用排序.

当对时间复杂度有要求时, 一个经验就是要看看使用 Hash, 经典以时间换空间.

### 自创算法

乍一看似乎时间复杂度不是O(n), 因为嵌套的while循环可能会导致更坏的时间复杂度. 但是, 仔细观察可以发现, 对于每个数字, 我们最多只会在两个方向上进行一次扩展 (向左和向右). 

```java
class Solution {
    public int longestConsecutive(int[] nums) {
        int res = 0;
        Map<Integer, Boolean> record = new HashMap<>();  // 记录所有数字和它是否被使用过 (用于构造连续数列)
        for (int num : nums) record.put(num, false);
        for (int cur : nums) {
            if (record.get(cur) == true) continue;  // 当该数字已经被使用过, 则不予考虑
            int len = 0;
            int tmp = cur;  // 临时变量 tmp, 用于遍历大于和小于 cur 的部分
            while (record.containsKey(tmp)) {  // 遍历 >= cur 的部分, 并在找到合适数字时, 将它标记成 true
                record.put(tmp, true);
                len ++;
                tmp ++;
            }
            tmp = cur - 1;
            while (record.containsKey(tmp)) {  // 遍历 < cur 的部分, 并在找到合适数字时, 将它标记成 true
                record.put(tmp, true);
                len ++;
                tmp --;
            }
            res = Math.max(res, len);
        }
        return res;
    }
}
```

这个算法性能更好

```go
func longestConsecutive(nums []int) int {
    var rec = make(map[int]bool)
    for _, v := range nums {
        rec[v] = false
    }

    ans := 0
    for _, num := range nums {
        if rec[num] {
            continue
        }

        len := 1
        rec[num] = true // 注意标记自己
        
        i:= num + 1
        for {
            _, exist := rec[i]
            if exist {
                len ++;
                rec[i] = true
                i ++;
            } else {
                break
            }
        }

        i = num - 1
        for {
            _, exist := rec[i]
            if exist {
                len ++;
                rec[i] = true
                i --;
            } else {
                break
            }
        }

        if len > ans {
            ans = len
        }
    }
    return ans
}
```



### 经典算法

算法关键是: 当 num 不能作为连续数列的最小值时, 就不予考虑; 当 num 可以作为连续数列的头部, 则使用 while 循环查看这一数列最长可以到达多长

```java
class Solution {
    public int longestConsecutive(int[] nums) {
        int res = 0;
        Set<Integer> set = new HashSet<>();
        for (int num : nums) set.add(num);

        for (int num : nums) {
            if (set.contains(num - 1)) continue;  // 这里是关键: 当 num 不能作为连续数列的最小值时, 就不予考虑
            int len = 0;
            int tmp = num;
            while (set.contains(tmp)) {
                len ++;
                tmp ++;
            }
            res = Math.max(res, len);
        }
        return res;
    }
}
```

### 总结

不管是哪种算法, 都使用了 hash, 以空间换时间.

```java
class Solution {
    private void fill (char[][] board, int i, int j) {  // 寻找最大连通区域
        int nr = board.length, nc = board[0].length;
        if (i == -1 || i == nr || j == -1 || j == nc || board[i][j] == 'X' || board[i][j] == '*') return;
        board[i][j] = '*';
        fill(board, i+1, j);
        fill(board, i-1, j);
        fill(board, i, j+1);
        fill(board, i, j-1);
    }

    public void solve(char[][] board) {
        int nr = board.length, nc = board[0].length;
        for (int i = 0; i < nr; i ++) {
            for (int j = 0; j < nc; j ++) {
                if (i == 0 || j == 0 || i == nr-1 || j == nc-1) {
                    if (board[i][j] == 'O') {
                        fill(board, i, j);
                    }
                }
            }
        }

        for (int i = 0; i < nr; i ++) {
            for (int j = 0; j < nc; j ++) {
                if (board[i][j] == '*') board[i][j] = 'O';
                else board[i][j] = 'X';
            }
        }
    }
```

```java
class Solution {
    private Map<Integer, Node> map = new HashMap<>();  // 要使用哈希表来记录克隆过的结点

    public Node cloneGraph(Node node) {
        if (node == null) return null;
        if (map.containsKey(node.val)) return map.get(node.val);  // 关键
        Node clone = new Node(node.val);
        map.put(clone.val, clone);
        for (Node neighbor: node.neighbors) {
            clone.neighbors.add(cloneGraph(neighbor));
        }
        return clone;
    }
}
```

## [140. Word Break II](https://leetcode.com/problems/word-break-ii/)

**纯回溯**

```java
class Solution {
    private List<String> res = new LinkedList<>();
    private List<String> path = new LinkedList<>();

    private boolean match(String word, String s, int start) {
        if (start + word.length() - 1 >= s.length()) return false;
        for (int i = 0; i < word.length(); i ++) {
            if (word.charAt(i) != s.charAt(start + i)) return false;
        }
        return true;
    }

    private void backtrack(String s, List<String> wordDict, int start) {
        if (start == s.length()) {
            res.add(path.stream().collect(Collectors.joining(" ")));
            return;
        }

        for (String word: wordDict) { // 可以使用 s.startWith(i, word) 来剪枝优化
           if (!match(word, s, start)) continue;
           path.add(word);
           backtrack(s, wordDict, start + word.length());
           path.removeLast();
        }
    }


    public List<String> wordBreak(String s, List<String> wordDict) {
        backtrack(s, wordDict, 0);
        return res;
    }
}
```

> 算法分析
>
> 假设 `s.length() = m`, `wordDict.size() = n`
>
> 每个 start 位置, 最多有 $O(n)$ 个可能的分支
>
> 整体的递归深度最坏是 $O(m)$
>
> 以上产生的复杂度为 $O(n^m)$, 再加上路径拼接等操作, 总体复杂度为 $O(m \cdot n ^ m)$

**倒序回溯 + 记忆化搜索**

```java
class Solution {
    public List<String> wordBreak(String s, List<String> wordDict) {
        this.s = s;
        this.label = canBuild(s, wordDict); // 含义是, s:[0, i]能否由wordDict构成
        this.wordDict = wordDict;
        this.ans = new ArrayList<>();
        this.path = new ArrayList<>();
        backtrack(s.length() - 1); // 倒着搜索, 利用 label 数组
        return ans;
    }

    private String s;
    private boolean[] label;
    private List<String> wordDict;
    private List<String> ans;
    private List<String> path;
    private void backtrack(int idx) {
        if (idx < 0) {
            List<String> tmp = new ArrayList<>(path);
            Collections.reverse(tmp);
            ans.add(String.join(" ", tmp));
            return;
        }
        if (label[idx]) { // 如果 s: [0, idx] 不能由 wordDict 构成, 则不找了
            for (String word : wordDict) {
                int cutStart = idx - word.length() + 1;
                if (cutStart < 0) continue;
                String cut = s.substring(cutStart, idx + 1);
                if (word.equals(cut)) {
                    path.add(word);
                    backtrack(cutStart - 1);
                    path.removeLast();
                }
            }
        }
    }

    private boolean[] canBuild(String s, List<String> wordDict) {
        boolean[] ret = new boolean[s.length()];
        for (int i = 0; i < s.length(); i ++) {
            for (String word : wordDict) {
                if (i + 1 < word.length() || ret[i]) continue;
                if (i + 1 == word.length()) ret[i] = word.equals(s.substring(0, i + 1));
                else if (i + 1 > word.length()) ret[i] = word.equals(s.substring(i - word.length() + 1, i + 1)) && ret[i - word.length()];
            }
        }
        return ret;
    }
}
```

>**核心思想**
>
>- 使用一个**记忆化数组 `label`**，其中 `label[i]` 表示 `s[0:i]` 能否由 `wordDict` 构成。
>- 如果 `label[idx] == false`，则跳过不必回溯的路径，**减少不必要的搜索**。
>- 采用从后往前的**倒序回溯**方法，尝试对 `s` 的末尾进行切分。
>- 当搜索到 `idx < 0` 时，表示已经从末尾回溯到了字符串的起始位置，路径可行，加入 `ans`。
>
>------
>
>**时间复杂度分析**
>
>1. **canBuild 的时间复杂度**
>   - `canBuild` 函数是一个**动态规划**的实现，类似于“单词拆分 I (Word Break I)”的解法。
>   - 外层是一个长度为 `m` 的循环，内层是**wordDict 中的每个单词**的比较。
>   - 假设 `wordDict` 中的单词个数为 `n`，每次 `substring` 操作的时间复杂度为 `O(|word|)`，总的时间复杂度为： O(m⋅n⋅∣word∣)O(m \cdot n \cdot |word|)O(m⋅n⋅∣word∣)
>   - 如果 `wordDict` 中的单词长度为常数，那么可以简化为： O(m⋅n)O(m \cdot n)O(m⋅n)
>2. **回溯部分的时间复杂度**
>   - `backtrack` 函数中，使用了 `label[idx]` 来**提前剪枝**，减少不必要的搜索。
>   - 理想情况下，回溯的深度是 `O(m)`，每层的分支是 `O(n)`，但**比第一个算法少很多不必要的递归**。
>   - 每个 `backtrack` 都只在**可行的起始位置**递归，而不是尝试所有分支，分支数大大减少。
>3. **路径拼接的时间复杂度**
>   - 由于 `Collections.reverse(tmp)` 和 `String.join(" ", tmp)` 都是 O(m) 级别的操作。
>   - 假设每次回溯路径的长度为 m（即拼接路径的长度），路径拼接的总时间复杂度为： O(m⋅k)O(m \cdot k)O(m⋅k) 其中 k 是路径的数量。
>
>------
>
>**时间复杂度总结**
>
>- **动态规划的复杂度**： $O(m⋅n)$
>- **回溯的复杂度**：
>  最坏情况下是指数级的，但由于使用了 `label` 来**剪枝**，实际的搜索空间大大减少。 $O(2^m)$
>- **路径拼接的复杂度**： $O(m \cdot k)$
>- **总时间复杂度**： $O(m \cdot n + 2^m \cdot m)$ 这个时间复杂度比第一个算法的 `O(n^m \cdot m)` **要小很多**，因为**有效的剪枝减少了搜索的分支数量**。

注意合理拆分功能

```java
class LRUCache {
    private class Node {
        int key;
        int val;
        Node next;
        Node prev;
        Node(int k, int v) {
            key = k; val = v;
        }
    }
    private Node head;
    private Node tail;
    private Map<Integer, Node> map;
    private int cap;
    private int sz;
    private void insertTail(Node node) {
        Node rest = tail.prev;
        node.next = tail;
        node.prev = rest;
        rest.next = node;
        tail.prev = node;
    }
    private void delNode(Node del) {
        if (del != head && del != tail) {
            Node prev = del.prev;
            Node next = del.next;
            prev.next = next;
            next.prev = prev;
        }
    }
    private void moveTail(Node node) {
        delNode(node);
        insertTail(node);
    }

    public LRUCache(int capacity) {
        cap = capacity;
        sz = 0;
        head = new Node(0, 0);
        tail = new Node(0, 0);
        map = new HashMap<>();
        head.next = tail;
        tail.prev = head;
    }
    
    public int get(int key) {
        if (!map.containsKey(key)) return -1;
        else {
            Node node = map.get(key);
            moveTail(node);
            return node.val;
        }
    }
    
    public void put(int key, int value) {
        if (map.containsKey(key)) {
            Node node = map.get(key);
            node.val = value;
            moveTail(node);
        } else {
            Node node = new Node(key, value);
            map.put(key, node);
            insertTail(node);
            if (sz < cap) sz ++;
            else {
                Node del = head.next;
                map.remove(del.key);
                delNode(del);
            }
        }
    }
}
```

```java
class Solution {
    private void insert(ListNode dummy, int val) {
        ListNode l = dummy, r = dummy.next;
        while (r != null) {  // 搜索插入位置可以使用二分搜索
            if (l.val <= val && val < r.val) break;
            r = r.next; l = l.next;
        }
        ListNode newNode = new ListNode(val);
        l.next = newNode; newNode.next = r;
    }

    public ListNode insertionSortList(ListNode head) {
        ListNode dummy = new ListNode(-9999);
        while (head != null) {
            insert(dummy, head.val);
            head = head.next;
        }
        return dummy.next;
    }
}
```

```java
class Solution {
    public ListNode sortList(ListNode head) {
        if (head == null || head.next == null) return head;
        
        ListNode slow = head, fast = head.next; // 这里不能设 slow = head, fast = head, 在head只有两个元素时, 这样的设法会造成无限递归 !!!
        while (fast != null && fast.next != null) {
            slow = slow.next;
            fast = fast.next.next;
        }
		
        ListNode l1 = head;
        ListNode l2 = slow.next;
        slow.next = null; // 注意置 null
        
        ListNode sorted1 = sortList(l1);
        ListNode sorted2 = sortList(l2);
		
        /* 合并 */
        ListNode ans = new ListNode(0);
        ListNode p = ans;
        while (sorted1 != null || sorted2 != null) {
            if (sorted2 == null || (sorted1 != null && sorted1.val <= sorted2.val) ) {
                ListNode nxt = sorted1.next;
                sorted1.next = null; // 注意置 null
                p.next = sorted1;
                sorted1 = nxt;
            } else if (sorted1 == null || (sorted2 != null && sorted1.val > sorted2.val) ) {
                ListNode nxt = sorted2.next;
                sorted2.next = null; // 注意置 null
                p.next = sorted2;
                sorted2 = nxt;
            }
            p = p.next;
        }
        return ans.next;
    }
}
```
