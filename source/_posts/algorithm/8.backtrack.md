# Fundamental

## 全排列

> 给定 `nums` (里面元素两两不同), 求 `nums` 构成的全排列

#### C++

使用 `int` 的 bits 位来记录使用情况 (因为`nums.size()` 一般不会很大, 所以可以这样做)

```c++
#include <bits/stdc++.h>
using namespace std;

class Solution {
    void backtrack(
        vector<vector<int>>& paths, 
        vector<int>& path, 
        vector<int>& nums, 
        int used
    ) {
        if (path.size() == nums.size()) { paths.push_back(path); return; }
        for (int i = 0; i < nums.size(); i ++) {
            if (!(used & (1 << i))) {
                path.push_back(nums[i]);
                backtrack(paths, path, nums, used | (1 << i));
                path.pop_back();
            }
        }
    }
public:
    vector<vector<int>> permute(vector<int>& nums) {
        vector<vector<int>> paths;
        vector<int> path;
        backtrack(paths, path, nums, 0);
        return paths;
    }
};

int main() {
    Solution s;
    vector<int> nums{0, 1, 2};
	vector<vector<int>> paths = s.permute(nums);
    for (auto& path : paths) {
        for (int num : path) cout << num << " ";
        cout << endl;
    }
}
```

时间复杂度: $O(n!), A^n_n$

空间复杂度: `used` $O(1)$ + `paths` $O(n * n!)$ = $O(n * n!)$



## 排列

> 给定 `nums` (里面元素两两不同), 求 `nums` 构成的数量为 k 的排列

#### C++

由于 `nums.size()` 可能较大, 因此不考虑使用 `int` 的 bits 位来记录状态, 而是采用 `vector<bool>`

```c++
#include <bits/stdc++.h>
using namespace std;

class Solution {
    void backtrack(
        vector<vector<int>>& paths, 
        vector<int>& path, 
        vector<int>& nums, 
        int k,
        vector<bool>& used
    ) {
        if (path.size() == k) { paths.push_back(path); return; }
        for (int i = 0; i < nums.size(); i ++) {
            if (!used[i]) {
                path.push_back(nums[i]);
                used[i] = true;
                backtrack(paths, path, nums, k, used);
                used[i] = false;
                path.pop_back();
            }
        }
    }
public:
    vector<vector<int>> permute(vector<int>& nums, int k) {
        vector<vector<int>> paths;
        vector<int> path;
        vector<bool> used(nums.size(), false);
        backtrack(paths, path, nums, k, used);
        return paths;
    }
};

int main() {
    Solution s;
    vector<int> nums{0, 1, 2, 3};
	vector<vector<int>> paths = s.permute(nums, 3);
    for (auto& path : paths) {
        for (int num : path) cout << num << " ";
        cout << endl;
    }
    /*
    0 1 2 
    0 1 3 
    0 2 1 
    0 2 3 
    0 3 1 
    0 3 2 
    1 0 2 
    1 0 3 
    1 2 0 
    1 2 3 
    1 3 0 
    1 3 2 
    2 0 1 
    2 0 3 
    2 1 0 
    2 1 3 
    2 3 0 
    2 3 1 
    3 0 1 
    3 0 2 
    3 1 0 
    3 1 2 
    3 2 0 
    3 2 1 
    */
}
```

时间复杂度: $O(\frac{n!}{(n - k)!}), A^k_n$

空间复杂度: $O(k * \frac{n!}{(n - k)!})$



## 组合

> 给定 `nums` (里面元素两两不同), 求 `nums` 构成的 k 个元素的排列

#### C++

写法1 (使用 start 来更新递归)

```c++
#include <bits/stdc++.h>
using namespace std;

class Solution {
    void backtrack(
        vector<vector<int>>& paths, 
        vector<int>& path, 
        vector<int>& nums, 
        int k,
        int start
    ) {
        if (path.size() == k) { paths.push_back(path); return; }
        for (int i = start; i <= nums.size() - (k - path.size()); i ++) {
            path.push_back(nums[i]);
            backtrack(paths, path, nums, k, i + 1);
            path.pop_back();
        }
    }
public:
    vector<vector<int>> combine(vector<int>& nums, int k) {
        vector<vector<int>> paths;
        vector<int> path;
        backtrack(paths, path, nums, k, 0);
        return paths;
    }
};

int main() {
    Solution s;
    vector<int> nums{0, 1, 2, 3};
	vector<vector<int>> paths = s.combine(nums, 3);
    for (auto& path : paths) {
        for (int num : path) cout << num << " ";
        cout << endl;
    }
    /*
    0 1 2 
    0 1 3 
    0 2 3 
    1 2 3 
    */
}
```

时间复杂度: $O(\frac{n!}{k!(n - k)!}), C^k_n$

空间复杂度: $O(k * \frac{n!}{k!(n - k)!})$



写法2 (类似填字类问题, 依次考虑选取还是不选取 `nums[idx]`)

```c++
#include <bits/stdc++.h>
using namespace std;

class Solution {
    void backtrack(
        vector<vector<int>>& paths, 
        vector<int>& path, 
        vector<int>& nums, 
        int k,
        int idx
    ) {
        if (idx + k - path.size() > nums.size()) return; // 剪枝
        if (path.size() == k) { paths.push_back(path); return; }
        path.push_back(nums[idx]); // 填入 nums[idx]
        backtrack(paths, path, nums, k, idx + 1);
        path.pop_back(); // 不填 nums[idx]
        backtrack(paths, path, nums, k, idx + 1);
    }
public:
    vector<vector<int>> combine(vector<int>& nums, int k) {
        vector<vector<int>> paths;
        vector<int> path;
        backtrack(paths, path, nums, k, 0);
        return paths;
    }
};

int main() {
    Solution s;
    vector<int> nums{0, 1, 2, 3};
	vector<vector<int>> paths = s.combine(nums, 3);
    for (auto& path : paths) {
        for (int num : path) cout << num << " ";
        cout << endl;
    }
    /*
    0 1 2 
    0 1 3 
    0 2 3 
    1 2 3 
    */
}
```



## 子集

> 相当于求所有可能长度的排列
>
> 给定 `nums` (里面元素两两不同), 求 `nums` 构成的所有子集

#### C++

写法1

```c++
#include <bits/stdc++.h>
using namespace std;

class Solution {
    void backtrack(
        vector<vector<int>>& paths, 
        vector<int>& path, 
        vector<int>& nums, 
        int start
    ) {
        paths.push_back(path);
        for (int i = start; i < nums.size(); i ++) {
            path.push_back(nums[i]);
            backtrack(paths, path, nums, i + 1);
            path.pop_back();
        }
    }
public:
    vector<vector<int>> subset(vector<int>& nums) {
        vector<vector<int>> paths;
        vector<int> path;
        backtrack(paths, path, nums, 0);
        return paths;
    }
};

int main() {
    Solution s;
    vector<int> nums{0, 1, 2};
	vector<vector<int>> paths = s.subset(nums);
    for (auto& path : paths) {
        cout << "[ ";
        for (int num : path) cout << num << " ";
        cout << "]" << endl;
    }
    /*
    [ ]
    [ 0 ]
    [ 0 1 ]
    [ 0 1 2 ]
    [ 0 2 ]
    [ 1 ]
    [ 1 2 ]
    [ 2 ]
    */
}
```

时间复杂度: [节点数: $O(2^n), 2^n = C^n_0 + ... + C^n_n$] * [每一个节点都要复制 `path`: $O(n)$] = $O(n * 2^n)$

空间复杂度: $O(n * 2 ^ n)$



写法2 (填字类写法)

```c++
#include <bits/stdc++.h>
using namespace std;

class Solution {
    void backtrack(
        vector<vector<int>>& paths, 
        vector<int>& path, 
        vector<int>& nums, 
        int idx
    ) {
        if (idx == nums.size()) return;
        path.push_back(nums[idx]);
        paths.push_back(path); // 注意, 在填入新的元素之后收集, 不能在 push_back 之前无条件收集, 不然会重复收集
        backtrack(paths, path, nums, idx + 1);
        path.pop_back();
        backtrack(paths, path, nums, idx + 1);
    }
public:
    vector<vector<int>> subset(vector<int>& nums) {
        vector<vector<int>> paths;
        paths.push_back({}); // 要提前加入空集合
        vector<int> path;
        backtrack(paths, path, nums, 0);
        return paths;
    }
};

int main() {
    Solution s;
    vector<int> nums{0, 1, 2};
	vector<vector<int>> paths = s.subset(nums);
    for (auto& path : paths) {
        cout << "[ ";
        for (int num : path) cout << num << " ";
        cout << "]" << endl;
    }
    /*
    [ ]
    [ 0 ]
    [ 0 1 ]
    [ 0 1 2 ]
    [ 0 2 ]
    [ 1 ]
    [ 1 2 ]
    [ 2 ]
    */
}
```





## 分割所有子串 (子数组)

> 相当于考查分位点 1, 2, ..., n, 每个分位点可选可不选, , 每一组选取就构成了字符串一种切分方法, 于是就转化成一个求子集的问题
>
> 给定一个字符串 `str`, 求出它的所有分割方法

#### C++

```c++
#include <bits/stdc++.h>
using namespace std;

class Solution {
    void backtrack(
        vector<vector<string>>& paths, 
        vector<string>& path, 
        string& str, 
        int start
    ) {
        if (start == str.size()) { paths.push_back(path); return; }
        for (int i = start; i < str.size(); i ++) {
            path.push_back(str.substr(start, i - start + 1));
            backtrack(paths, path, str, i + 1);
            path.pop_back();
        }
    }
public:
    vector<vector<string>> splitStr(string& str) {
        vector<vector<string>> paths;
        vector<string> path;
        backtrack(paths, path, str, 0);
        return paths;
    }
};

int main() {
    Solution s;
    string str = "abc";
    vector<vector<string>> allSplits = s.splitStr(str);
    for (auto& split : allSplits) {
        for (int i = 0; i < split.size(); i ++) {
            cout << split[i];
            if (i < split.size() - 1) cout << " | ";
        }
        cout << endl;
    }
    /*
    a | b | c
    a | bc
    ab | c
    abc
    */
}
```

时间复杂度: [节点数: $O(2^n), 2^n = C^n_0 + ... + C^n_n$] * [每一个节点都要执行 `substr` : $O(n)$] = $O(n * 2^n)$

空间复杂度: $O(n * 2 ^ n)$

#### Java

```java
class Solution {
    private void backtrack(List<List<Integer>> ans, List<Integer> path, int n, int k, int startNum) {
        if (path.size() == k) {
            ans.add(new ArrayList<>(path));
            return;
        }
        for (int i = startNum; i <= n - (k - path.size()) + 1; i ++) {
            path.add(i);
            backtrack(ans, path, n, k, i + 1);
            path.removeLast();
        }
    }

    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> ans = new ArrayList<>();
        List<Integer> path = new ArrayList<>();
        backtrack(ans, path, n, k, 1);
        return ans;
    }
}

func letterCombinations(digits string) []string {
    tab := []string {
        "",
        "",
        "abc",
        "def",
        "ghi",
        "jkl",
        "mno",
        "pqrs",
        "tuv",
        "wxyz",
    }
    paths := []string{}
    if len(digits) == 0 {
        return paths
    }
    path := ""

    var backtrack func()
    backtrack = func() {
        idx := len(path)
        if idx == len(digits) {
            paths = append(paths, path)
            return
        }
        digit := (int)(digits[idx] - '0')
        for _, c := range tab[digit] {
            path += string(c)
            backtrack()
            path = path[:len(path) - 1]
        }
    }

    backtrack()
    return paths
}

class Solution {
    private void backtrack(List<List<Integer>> ans, int[] candidates, List<Integer> path, int pathSum, int target, int idx) {
        if (pathSum > target) return;
        if (pathSum == target) {
            ans.add(new ArrayList<>(path));
            return;
        }
        for (int i = idx; i < candidates.length; i ++) {
            path.addLast(candidates[i]);
            backtrack(ans, candidates, path, pathSum + candidates[i], target, i);
            path.removeLast();
        }
    }
    public List<List<Integer>> combinationSum(int[] candidates, int target) {
        List<List<Integer>> ans = new ArrayList<>();
        backtrack(ans, candidates, new ArrayList<>(), 0, target, 0);
        return ans;
    }
}
```

## [40. Combination Sum II](https://LeetCode.com/problems/combination-sum-ii/)

```
candidates = [1, 1, 2], target = 3
idx        =  0  1  3

去重: 当 idx = 0 时, 下层的递归就会搜出以 1(0) 开头的所以组合, 也就是: 
	[1(0), 1(1)], 
	[1(0), 1(1), 2(3)], 
	[1(0), 2(3)], 
取得结果 [1(0), 2(3)]
如果不去重, 就会搜出 [1(1), 2(3)], 它也会加入进结果
```

#### Java

排序去重

```java
class Solution {
    private void backtrack(List<List<Integer>> ans, int[] candidates, int target, List<Integer> path, int sum, int start) {
        if (target < sum) return;
        if (target == sum) {
            ans.add(new ArrayList<>(path));
            return;
        }
        for (int i = start; i < candidates.length && sum + candidates[i] <= target; i ++) { // 由于排过序, child1 <= child2 <= ... 如果 child1 + curSum > target 就不需要在搜索 child2, 3,...了
            if (i > start && candidates[i] == candidates[i-1]) continue; // 跳过同一树层使用过的元素
            path.add(candidates[i]);
            backtrack(ans, candidates, target, path, sum + candidates[i], i + 1);
            path.removeLast();
        }
    }
    public List<List<Integer>> combinationSum2(int[] candidates, int target) {
        List<List<Integer>> ans = new ArrayList<>();
        Arrays.sort(candidates);
        backtrack(ans, candidates, target, new ArrayList<>(), 0, 0);
        return ans;
    }
}
```

        this.path = new ArrayList<>();

    private String s;
    private boolean[][] label;
    private List<List<String>> ans;
    private List<String> path;

    private void backtrack(int idx) {
        if (idx == s.length()) {
            ans.add(new ArrayList<>(path));
            return;
        }
        for (int i = idx; i < s.length(); i ++) { // s:[idx, i]
            if (label[idx][i]) {
                path.add(s.substring(idx, i + 1));
                backtrack(i + 1);
                path.removeLast();
            }
        }

    private boolean[][] partitionMatrix(String s) {
        boolean[][] ret = new boolean[s.length()][s.length()];
        for (int i = s.length() - 1; i > -1; i --) {
            for (int j = i; j < s.length(); j ++ ) {
                if (i + 1 >= j) ret[i][j] = s.charAt(i) == s.charAt(j);
                else ret[i][j] = ret[i + 1][j - 1] && s.charAt(i) == s.charAt(j);
            }
        }
        return ret;
    }
}

class Solution {
    private boolean isPartition(String s) {
        int left = 0, right = s.length() - 1;
        while (left <= right) {
            if (s.charAt(left) != s.charAt(right)) return false;
            left ++; right --;
        }
        return true;
    }
    private void backtrack(String s, List<List<String>> res, List<String> path, int start) {
        if (start == s.length()) {
            res.add(new ArrayList<>(path));

            if (!isPartition(sub)) continue;

            backtrack(s, res, path, i + 1);
            path.removeLast();
        }
    }

func backtrack(label [][]bool, ans *[][]string, path *[]string, s string, idx int) {
    if idx == len(s) {
        // 深拷贝 path，避免回溯导致的 path 被覆盖
        temp := make([]string, len(*path))
        copy(temp, *path)
        *ans = append(*ans, temp)
        return
    }

    for i := idx; i < len(s); i++ {
        if label[idx][i] {
            *path = append(*path, s[idx:i+1]) // 只在需要时创建子字符串
            backtrack(label, ans, path, s, i+1)
            *path = (*path)[:len(*path)-1] // 回溯
        }
    }
}

func partitionLabel(s string) [][]bool {
    n := len(s)
    ret := make([][]bool, n)
    for i := range ret {
        ret[i] = make([]bool, n)
    }

    for i := n - 1; i >= 0; i-- {
        for j := i; j < n; j++ {
            if j - i <= 1 {
                ret[i][j] = s[i] == s[j]
            } else {
                ret[i][j] = s[i] == s[j] && ret[i+1][j-1]
            }
        }
    }
    return ret
}

        List<List<String>> tickets, 
        List<String> path,
        Map<String, Map<String, Integer>> map

        if (path.size() == tickets.size() + 1) return true;
        String prePort = path.getLast(); // 上一个 port

                String curPort = e.getKey(); // 现在要去的 port
                int cnt = e.getValue();
                if (cnt > 0) {

                    if (backtrack(tickets, path, map)) return true;
                    path.removeLast();
                    e.setValue(cnt);
                }
            }
        }

class Solution {
    private int res;
    private void backtrack(int[] path, int th) {
        if (th == path.length) {
            res ++;
            return;
        }
        // 确定 path[th] 0, 1, ..., path.length - 1
        for (int i = 0; i < path.length; i++) {  // 在棋盘 th 行处, 尝试把棋子放在 i 位置上
            path[th] = i;
            if (!isValid(path)) {
                path[th] = -1;
                continue;
            }
            backtrack(path, th + 1);
            path[th] = -1;
        }
    }
    private boolean isValid(int[] path) {
        for (int i = 0; i < path.length; i ++) {
            if (path[i] == -1) break;
            for (int j = 0; j < i; j ++) {
                if (path[j] == path[i]) return false;
                if (Math.abs(i - j) == Math.abs(path[i] - path[j])) return false;
            }
        }
        return true;
    }

    public int totalNQueens(int n) {
        int[] path = new int[n];
        Arrays.fill(path, -1);
        backtrack(path, 0);
        return res;
    }
}

    private fun isValid(pos: Int): Boolean { // 可以省略 lineIdx, 因为每一次添加必然是在path最后
        for (i in path.indices) {
            if (pos == path[i]) return false
            if (kotlin.math.abs(pos - path[i]) == path.size - i) return false
        }
        return true
    }

    private fun backtrack(): Unit {
        if (path.size == n) { 
            ans ++
            return
        }
        for (pos in 0 ..< n) {
            if (!isValid(pos)) continue
            path.add(pos)
            backtrack()
            path.removeLast()
        }
    }

    fun totalNQueens(n: Int): Int {
        this.n = n
        backtrack()
        return ans
    }
}
